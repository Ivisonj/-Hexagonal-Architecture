model Users {
  id       String     @id @default(uuid())
  name     String
  email    String     @unique
  password String
  articles Articles[]
}

model Articles {
  id        String   @id @default(uuid())
  title     String
  subtitle  String
  createdAt DateTime @default(now())
  content   String
  userId    String
  user      Users    @relation(fields: [userId], references: [id])
}

//---COMANDOS
1 - npx prisma init

2 - criar schema

3 - npx prisma generate

4 - npx prisma migrate dev --name init
